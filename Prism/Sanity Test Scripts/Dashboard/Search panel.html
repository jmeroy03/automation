Search panel
pause 5000
readCSV file://C:\Automation\Prism\Data\Header\search_panel.csv
store KEY_ENTER
for $row=3; $row<=9; $row++
storeCellValue search_options ${row},1
storeCellValue user_input ${row},2
storeCellValue page_url ${row},3
storeCellValue layout_id ${row},4
storeCellValue id_element ${row},5
echo ${search_options}
clickAt css=div > b
sendKeys css=input[type="text"] ${search_options}${KEY_ENTER}
verifyText css=a.chosen-single > span ${search_options}
verifyVisible css=a.chosen-single > span
sendKeys id=search_box ${user_input}
click css=div.input-group-btn > button.btn.btn-default
pause 10000
verifyElementNotPresent //div[@id='content']/div/div/div/div/div/h1
verifyNotLocation http://qaprism.digitalroom.com/#/404
verifyLocation ${page_url}
verifyText ${id_element} ${layout_id}
verifyElementPresent ${id_element}
verifyVisible ${id_element}
endFor
