Customers - Submit Form with checking
label EMAIL VERIFICATION
click xpath=(//button[@type='submit button'])[2]
pause 10000
storeElementPresent css=div.textoFoto > p alert_notification
if ${alert_notification}==true
storeText css=div.textoFoto > p alert_message
store Email address already exists error_message
store Customer successfully created. customer_created_message
storeEval $alert_message==$error_message customer_exists
storeEval $alert_message==$customer_created_message customer_created
if ${customer_exists}==true
type //input[@type='email']
store edsel.q.uprinting. enter_email
call random_email email=$enter_email
echo ${_result}
sendKeys //input[@type='email'] ${_result}
storeValue //input[@type='email'] stored_email
click //div/div[2]/span/i
goto EMAIL VERIFICATION
elseIf ${customer_created}==true
echo "Customer Successfully Created"
else
getEval LOG.error("Other error occured upon submiting form");
storeEval alert ('Check logs for error');
exitTest
endIf
endIf
pause 30000
